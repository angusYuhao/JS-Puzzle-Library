Small Grid Puzzle

A simple 2 x 2 puzzle with a 1 x 4 tray

                    
                        const canvasContainer = document.querySelector('#canvas')
                        const trayContainer = document.querySelector('#tray')
                        
                        canvasContainer.addEventListener('puzzleUpdated', function (e) {
                            if (e.detail.puzzleStatus) {
                                let popup = document.querySelector(".completeNotification")
                                popup.addEventListener("click", () => { popup.style.display = "none" })
                                popup.style.display = "block"
                            }
                        })
                        
                        const orderMap = [
                            [0, 1, 2, 3, 4, 5, 6, 7, 8],
                        ]
                        
                        const puzzle = new ImagePuzzle("grid")
                        puzzle.bindImage("./soul_of_cinder.jpg")
                        puzzle.createGridCanvas(canvasContainer, 3, 3, 4, 
                                                "grey", "white", "Soul of Cinder", 
                                                "rgba(255, 59, 59, 0.5)", "4px", "rgba(25, 25, 25, 1)", "4px")
                        puzzle.createTray(trayContainer, 1, 9, 4, orderMap, false, 
                                                "cornflowerblue", "white", "Title Here", 
                                                "rgba(255, 59, 59, 0.5)", "4px", "rgba(25, 25, 25, 1)", "4px")

Scatter Puzzle

A "fill in the blanks" example

                    
                        const canvasContainer = document.querySelector('#canvas')
                        const trayContainer = document.querySelector('#tray')
                        
                        canvasContainer.addEventListener('puzzleUpdated', function (e) {
                            if (e.detail.puzzleStatus) {
                                let popup = document.querySelector(".completeNotification")
                                popup.addEventListener("click", () => { popup.style.display = "none" })
                                popup.style.display = "block"
                            }
                        })
                        
                        let orderMap = [
                            [0, 1, 2]
                        ]
                        
                        let slots = [
                            { top: "41%", left: "22.7%", height: "9.5%", width: "10.3%"},
                            { top: "40%", left: "74.9%", height: "9%", width: "20.4%"},
                            { top: "74.8%", left: "59.9%", height: "9.1%", width: "16.6%"}
                        ]
                        
                        let idArray = [0, 1, 2]
                        
                        let imgArray = ["./Apollo_engine.jpg", "./Apollo_command.jpg", "Apollo_lunar.jpg"]
                        
                        const puzzle = new ImagePuzzle("scatter")
                        puzzle.bindBackgroundImage("./Apollo.jpg")
                        puzzle.createScatterCanvas(canvasContainer, slots, 
                                                    "rgba(25, 25, 25, 1)", "white", "Apollo Diagram", 
                                                    "rgba(50, 50, 50, 0.5)", "4px", "grey", "4px")
                        
                        puzzle.createScatterMatchings(idArray, imgArray)
                        puzzle.createTray(trayContainer, 1, 3, 4, orderMap, true, 
                                                    "rgba(25, 25, 25, 1)", "white", "Apollo Diagram Puzzle Tray", 
                                                    "rgba(50, 50, 50, 0.5)", "4px", "grey", "4px")
